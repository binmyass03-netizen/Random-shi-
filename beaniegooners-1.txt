local stealEnabled = false
local stealCoroutine
local function getHRP()
    local char = LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()
    return char:WaitForChild("HumanoidRootPart", 5)
end
local function getPromptPart(prompt)
    local par = prompt.Parent
    if par:IsA("BasePart") then return par end
    if par:IsA("Model") then return par.PrimaryPart or par:FindFirstChildWhichIsA("BasePart") end
    if par:IsA("Attachment") then return par.Parent end
    return par:FindFirstChildWhichIsA("BasePart", true)
end
local function findStealPrompt(hrp)
    local nearest, minD = nil, math.huge
    local plots = workspace:FindFirstChild("Plots"); if not plots then return nil end
    for _, d in pairs(plots:GetDescendants()) do
        if d:IsA("ProximityPrompt") and d.Enabled and d.ActionText == "Steal" then
            local pt = getPromptPart(d)
            if pt then
                local dist = (hrp.Position - pt.Position).Magnitude
                if dist < minD then minD = dist; nearest = d end
            end
        end
    end
    return nearest
end
local function triggerPrompt(p)
    if not p or not p:IsDescendantOf(workspace) then return end
    p.MaxActivationDistance = 9e9; p.RequiresLineOfSight = false; p.ClickablePrompt = true
    local ok = pcall(function() fireproximityprompt(p, 9e9, 0.5) end)
    if not ok then pcall(function() p:InputHoldBegin(); task.wait(0.5); p:InputHoldEnd() end) end
end
local function stealLoop()
    while stealEnabled do
        local hrp = getHRP()
        if hrp then local p = findStealPrompt(hrp); if p then triggerPrompt(p) end
        else LocalPlayer.CharacterAdded:Wait() end
        task.wait(0.2)
    end
end
local function setStealing(state)
    if state == stealEnabled then return end
    stealEnabled = state
    if stealEnabled then
        stealCoroutine = coroutine.create(stealLoop)
        coroutine.resume(stealCoroutine)
    else stealCoroutine = nil end
end

local function fireClick(button)
    if not button then return end
    if firesignal then
        pcall(function()
            firesignal(button.MouseButton1Click)
            firesignal(button.MouseButton1Down)
            firesignal(button.Activated)
        end)
    else
        local x = button.AbsolutePosition.X + button.AbsoluteSize.X/2
        local y = button.AbsolutePosition.Y + button.AbsoluteSize.Y/2 + 58
        VIM:SendMouseButtonEvent(x, y, 0, true,  game, 0)
        VIM:SendMouseButtonEvent(x, y, 0, false, game, 0)
    end
end

local function isOnCooldown(cmd)
    local adminGui = PlayerGui:FindFirstChild("AdminPanel")
    if adminGui then
        local ok, res = pcall(function()
            local sf = adminGui.AdminPanel.Content.ScrollingFrame
            local b = sf:FindFirstChild(cmd)
            if b then local t = b:FindFirstChild("Timer"); if t then return t.Visible end end
            return false
        end)
        if ok and res then return true end
    end
    if not activeCooldowns[cmd] then return false end
    return (tick() - activeCooldowns[cmd]) < (COOLDOWNS[cmd] or 0)
end

local function runAdminCommand(targetPlayer, commandName)
    local adminGui = PlayerGui:WaitForChild("AdminPanel", 5)
    if not adminGui then return false end
    local ok, res = pcall(function()
        local contentScroll = adminGui.AdminPanel.Content.ScrollingFrame
        local cmdBtn = contentScroll:FindFirstChild(commandName)
        if not cmdBtn then return false end
        fireClick(cmdBtn); task.wait(0.05)
        local profilesScroll = adminGui.AdminPanel.Profiles.ScrollingFrame
        local playerBtn = profilesScroll:FindFirstChild(targetPlayer.Name)
        if not playerBtn then return false end
        fireClick(playerBtn); return true
    end)
    return ok and res
end
_G.runAdminCommand = runAdminCommand

local function spamAllCmds(plr)
    local count = 0
    for _, cmd in ipairs(ALL_CMDS) do
        if not isOnCooldown(cmd) then
            task.delay(count * 0.12, function()
                if runAdminCommand(plr, cmd) then activeCooldowns[cmd] = tick() end
            end)
            count += 1
        end
    end
    return count
end
